// å¹¶æŸ¥é›†ï¼ˆUnion-Findï¼‰ç»“æ„
class UnionFind {
public:
    vector<int> parent;

    UnionFind(int size) {
        parent.resize(size);
        for (int i = 0; i < size; ++i) {
            parent[i] = i;
        }
    }

    int find(int x) {
        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

    void unionSet(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX != rootY) {
            parent[rootX] = rootY;
        }
    }
};

// æ¯æ¡è¾¹çš„ç»“æ„

// Kruskalç®—æ³•æ„é€ æœ€å°ç”Ÿæˆæ ‘
vector<pair<int, int>> kruskal(int n, vector<Edge>& edges) {
    sort(edges.begin(), edges.end(), [](const Edge& a, const Edge& b) {
        return a.weight < b.weight;
    });

    UnionFind uf(n);

        int rootV = uf.find(edge.v);
        if (rootU != rootV) {
            uf.unionSet(rootU, rootV);
            mst.emplace_back(edge.u, edge.v);
        }
    }
    return mst;
}
